isi dari perencanaannya sebagai berikut :

buat tabel pengeluaran:
id integer auto increment
nama varchar 255 not null
kategori_id int
tgl datetime
created_at timestamp default current_timestamp

buatkan api untuk get pengeluaran
endpoint : GET /api/pengeluaran

Response Body Success : 
{
    "data": []
}

buatkan api untuk get pengeluaran dengan tahun bulan tanggal spesifik dan bisa hanya salah satu (tahun, bulan, tanggal)
endpoint : GET /api/pengeluaran/{tahun}/{tanggal}/{bulan}

Response Body Success : 
{
    "data": "OK"
}

buatkan api untuk tambah pengeluaran
endpoint : POST /api/pengeluaran

Request Body : 
{
    "nama": "Beli Makan Siang",
    "kategori_id": 3,
    "tgl": 2026-06-01
}

Response Body Success : 
{
    "data": "OK"
}


buatkan api untuk hapus pengeluaran
endpoint : DELETE /api/pengeluaran/{id}

Response Body Success : 
{
    "data": "OK"
}

{
    "error": "Pengeluaran Tidak Ada"
}
